\startluacode
function zhvert(arg)
    for c in arg:utfcharacters() do
        if #c >=3 then
            context.rotate({rotation='90'}, c)
            context([[\kern.2em]])
        else
            context([[\kern.5em ]])
        end
    end
end
\stopluacode
